def overlaid_histogram(data1, data1_name, data1_color, data2, data2_name, data2_color, x_label, y_label, title):
    # å½’ä¸€åŒ–æ•°æ®åŒºé—´ï¼Œå¯¹é½ä¸¤ä¸ªç›´æ–¹å›¾çš„bins
    max_nbins = 10
    #è®¡ç®—è¾¹ç•Œ
    data_range = [min(min(data1), min(data2)), max(max(data1), max(data2))]
    binwidth = (data_range[1] - data_range[0]) / max_nbins
    bins = np.arange(data_range[0], data_range[1] + binwidth, binwidth) # ç”Ÿæˆç›´æ–¹å›¾binsåŒºé—´

    # Create the plot
    _, ax = plt.subplots()
    ax.hist(data1, bins = bins, color = data1_color, alpha = 1, label = data1_name)
    ax.hist(data2, bins = bins, color = data2_color, alpha = 0.75, label = data2_name)
    ax.set_ylabel(y_label)
    ax.set_xlabel(x_label)
    ax.set_title(title)
    ax.legend(loc = 'best')
